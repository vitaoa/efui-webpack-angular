1. 单选框(radio):

    js+css全兼容：

    
	
	
	

    less:
.radio-group{
	.radio-item{
		input[type="radio"]{opacity: 0;filter: alpha(opacity=0);}
		label{padding:1px 0;background-image: url(../images/radio_icon_default.png);background-position: center left;background-repeat: no-repeat;padding-left: 30px;}
		.checked_label{background-image: url(../images/radio_icon_checked.png);}
	}
}

    jquery:
$('.radio-group .radio-item').each(function () {
	$(this).find('input').eq(0).prop('checked') && $(this).find('label').addClass('checked_label');
	$(this).find('label').click(function () {
		$(this).closest('.radio-item').siblings().find('label').removeClass('checked_label');
		!$(this).siblings('input').eq(0).prop('checked') && $(this).addClass('checked_label');
	})
})
